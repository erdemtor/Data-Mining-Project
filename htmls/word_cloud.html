def generate_wordcloud(decade, imagefile, outfile):
    text = ' '.join(data.Lyrics.tolist())
    alice_mask = np.array(Image.open(imagefile))
    stopwords = set(STOPWORDS)
    stopwords.add('im')
    stopwords.add('got')
    stopwords.add('oh oh')
    wc = WordCloud(background_color="white", max_words=300, mask=alice_mask, stopwords=stopwords)
    wc.generate(text)
    wc.to_file(outfile)
    plt.imshow(wc, interpolation='bilinear')
    plt.axis("off")
    plt.figure()
    plt.imshow(alice_mask, cmap=plt.cm.gray, interpolation='bilinear')
    plt.axis("off")
    plt.show()
