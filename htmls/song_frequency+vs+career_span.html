def plot(x, y, title, xlabel, ylabel):
    fig = plt.figure(figsize=(10,8))
    plt.scatter(x, y, s=30, c='b', marker="D", alpha=0.5)
    fig.suptitle(title, fontsize=20)
    plt.xlabel(xlabel, fontsize=18)
    plt.ylabel(ylabel, fontsize=16)
    plt.show()
